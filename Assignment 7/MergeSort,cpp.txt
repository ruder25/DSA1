#include<iostream>
using namespace std;

void merge(int a[],int b,int c,int r){
int n1 = c-b+1;
 int n2 = r-c;

int L[n1],R[n2];
for(int i =0; i<n1; i++){
     L[i] =a[b+i];
}
     for(int j=0; j<n2; j++){
        R[j] = a[c+1+j];
     }
    int i=0,j=0,k=b;
    while(i< n1 && j< n2){
    if(L[i] <= R[j]){ 
        a[k] = L[i];
         i++;
         }
    else{ 

        a[k] = R[j];
         j++;}
        k++;
        }
    while(i<n1){

         a[k]  =L[i];
          i++; 
          k++; 
        }
    while(j<n2){ 
        a[k] =R[j]; 
        j++; k++; 
    }
}

void mergesort(int a[],int l,int r){
   if(l>=r) return;
int m = l+(r-l)/2;
mergesort(a,l,m);
    mergesort(a,m+1,r);
 merge(a,l,m,r);
}

int main(){
int a[]={5,2,9,1,7,3};
int n=6;
mergesort(a,0,n-1);
for(int i=0;i<n;i++){
     cout<<a[i]<<" ";
}
    return 0;
}
